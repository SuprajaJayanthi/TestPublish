import pytest

def test_display_policies_due_for_renewal():
    # Verify that the system displays all policies due for renewal within 30 days
    policies = get_policies_due_for_renewal()
    assert len(policies) > 0
    assert all(policy['last_renewal_days] <= 30 for policies)
